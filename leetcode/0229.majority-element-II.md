Given an integer array of size n, find all elements that appear more than ⌊ n/3 ⌋ times.

Example 1:
Input: nums = [3,2,3]
Output: [3]

Example 2:
Input: nums = [1]
Output: [1]

Example 3:
Input: nums = [1,2]
Output: [1,2]
 
Constraints:
1 <= nums.length <= 5 * 10^4
-10^9 <= nums[i] <= 10^9
 
Follow up: Could you solve the problem in linear time and in O(1) space?

Solution: Boyer-Moore Voting Algorithm

#### Typescript
```ts
function majorityElement(nums: number[]): number[] {
    let num1 = null;
    let num2 = null;
    let freq1 = 0;
    let freq2 = 0;

    for(const num of nums) {
        if(num === num1) {
            freq1++;
        } else if(num === num2) {
            freq2++;
        } else if(freq1 === 0) {
            freq1 = 1;
            num1 = num
        } else if(freq2 === 0) {
            freq2 = 1;
            num2 = num
        } else {
            freq1--;
            freq2--;
        }
    }

    freq1 = 0;
    freq2 = 0;

    for(const num of nums) {
        if(num === num1) freq1++;
        if(num === num2) freq2++
    }

    const res = [];
    const treshold = Math.floor(nums.length / 3); 
    if(freq1 > treshold) res.push(num1);
    if(freq2 > treshold) res.push(num2);

    return res;
};
```

#### Go
```go
func majorityElement(nums []int) []int {
    freq1 := 0
    freq2 := 0
    var num1 int
    var num2 int

    for _, num := range(nums) {
        if num == num1 {
            freq1++
        } else if num == num2 {
            freq2++
        } else if freq1 == 0 {
            num1 = num
            freq1 = 1
        } else if freq2 == 0 {
            num2 = num
            freq2 = 1
        } else {
            freq1--
            freq2--
        }
    }

    freq1 = 0
    freq2 = 0

    for _, num := range(nums) {
        if num == num1 {
            freq1++
        } else if num == num2 {
            freq2++
        }
    }

    res := make([]int, 0)
    k := 3
    treshold := len(nums) / k

    if freq1 > treshold {
        res = append(res, num1)
    } 
    if freq2 > treshold {
        res = append(res, num2)
    }

    return res
}
```
